#include <sys/sem.h>

void collector_main() {
    printf("[Collector] 启动，开始监控文件: %s\n", LOG_FILE);

    // 1. 初始化共享内存与信号量
    int shmid = shmget(SHM_KEY, sizeof(struct shared_data), IPC_CREAT | 0666);
    struct shared_data *shm = (struct shared_data *)shmat(shmid, NULL, 0);
    
    int semid = semget(SEM_KEY, 1, IPC_CREAT | 0666);
    semctl(semid, 0, SETVAL, 1); // 初始可用

    // 2. 打开日志文件
    int fd = open(LOG_FILE, O_RDONLY);
    if (fd < 0) {
        perror("打开日志文件失败");
        exit(EXIT_FAILURE);
    }

    // 3. 断点续传记录（简单示例：直接跳到文件末尾或读取特定 offset）
    off_t offset = lseek(fd, 0, SEEK_END);
    char buffer[BUFFER_SIZE];

    while (1) {
        // 尝试读取新增内容
        ssize_t bytes_read = read(fd, buffer, BUFFER_SIZE - 1);
        
        if (bytes_read > 0) {
            buffer[bytes_read] = '\0';
            
            // 获取信号量，写入共享内存
            sem_p(semid);
            strncpy(shm->text, buffer, BUFFER_SIZE);
            printf("[Collector] 采集到新日志并写入共享内存，大小: %zd bytes\n", bytes_read);
            
            // 记录新的偏移量
            offset += bytes_read;
            sem_v(semid); // 释放信号量供 Processor 读取
        } else {
            // 没有新内容，短暂休眠
            usleep(500000); // 500ms
        }
    }

    shmdt(shm);
    close(fd);
    exit(0); // 替换原来的 return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <signal.h>
#include <sys/msg.h>
#include <sys/stat.h>

#include <string.h>

// 令牌桶控制
volatile sig_atomic_t tokens = MAX_TOKENS;

// SIGALRM 处理函数：定时补充令牌
void alarm_handler(int sig) {
    if (tokens < MAX_TOKENS) {
        tokens++;
    }
}

// 守护进程化函数 (需要补回这里)
void daemonize() {
    pid_t pid = fork();
    if (pid < 0) exit(EXIT_FAILURE);
    if (pid > 0) exit(EXIT_SUCCESS); // 父进程退出

    if (setsid() < 0) exit(EXIT_FAILURE); // 创建新会话

    pid = fork(); // 第二次 fork，防止重新获取控制终端
    if (pid < 0) exit(EXIT_FAILURE);
    if (pid > 0) exit(EXIT_SUCCESS);

    umask(0);
    chdir("/"); // 改变工作目录

    // 关闭标准文件描述符
    close(STDIN_FILENO);
    close(STDOUT_FILENO);
    close(STDERR_FILENO);
}

// 单实例锁检测 (需要补回这里)
void enforce_single_instance() {
    int fd = open(LOCK_FILE, O_RDWR | O_CREAT, 0640);
    if (fd < 0) exit(EXIT_FAILURE);

    struct flock fl;
    fl.l_type = F_WRLCK;
    fl.l_whence = SEEK_SET;
    fl.l_start = 0;
    fl.l_len = 0;

    if (fcntl(fd, F_SETLK, &fl) < 0) {
        if (errno == EACCES || errno == EAGAIN) {
            // 锁已被占用，说明已有实例运行
            exit(EXIT_FAILURE);
        }
    }
}

void outputer_main() {
    // 1. 守护进程化 & 单实例运行
    daemonize();
    enforce_single_instance();

    // 2. 注册信号处理与令牌桶定时器 (每 100ms 补充一个令牌)
    struct sigaction sa;
    sa.sa_handler = alarm_handler;
    sa.sa_flags = SA_RESTART;
    sigemptyset(&sa.sa_mask);
    sigaction(SIGALRM, &sa, NULL);

    struct itimerval timer;
    timer.it_value.tv_sec = 0;
    timer.it_value.tv_usec = 100000;
    timer.it_interval.tv_sec = 0;
    timer.it_interval.tv_usec = 100000;
    setitimer(ITIMER_REAL, &timer, NULL);

    // 3. 获取消息队列与打开目标文件
    int msgid = msgget(MSG_KEY, 0666);
    int out_fd = open(OUTPUT_FILE, O_WRONLY | O_CREAT | O_APPEND, 0666);
    struct msg_buffer message;

    // 4. 消费循环 (带精细化流量控制)
    while (1) {
        if (tokens > 0) {
            // 尝试接收消息 (非阻塞或阻塞，这里使用阻塞)
            if (msgrcv(msgid, &message, sizeof(message.msg_text), 1, 0) > 0) {
                // 消耗令牌
                tokens--;
                
                // 写入下游存储
                write(out_fd, message.msg_text, strlen(message.msg_text));
                write(out_fd, "\n", 1);
            }
        } else {
            // 没有令牌，等待定时器信号唤醒
            pause(); 
        }
    }

    close(out_fd);
    exit(0); // 替换原来的 return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/shm.h>
#include <sys/sem.h>
#include <sys/msg.h>
#include <sys/wait.h>
#include <time.h>
#include "loghawk.h"

// Worker 进程的核心处理逻辑
void worker_process(int worker_id, struct shared_data *shm, int semid, int msgid) {
    struct msg_buffer message;
    message.msg_type = 1;

    while (1) {
        sem_p(semid); // 互斥读取共享内存
        if (strlen(shm->text) > 0) {
            // 解析、过滤与格式化 (添加时间戳)
            time_t now = time(NULL);
            char *time_str = ctime(&now);
            time_str[strlen(time_str)-1] = '\0'; // 移除换行符

            snprintf(message.msg_text, sizeof(message.msg_text), "[%s] [Worker %d] %s", time_str, worker_id, shm->text);
            
            // 发送到消息队列
            msgsnd(msgid, &message, sizeof(message.msg_text), 0);
            printf("[Processor Worker %d] 处理并转发日志至消息队列.\n", worker_id);
            
            // 清空已读数据
            memset(shm->text, 0, BUFFER_SIZE);
        }
        sem_v(semid);
        usleep(100000); // 避免 CPU 空转
    }
}

void processor_main() {
    printf("[Processor] 启动，准备 Fork %d 个子进程...\n", WORKER_COUNT);

    int shmid = shmget(SHM_KEY, sizeof(struct shared_data), 0666);
    struct shared_data *shm = (struct shared_data *)shmat(shmid, NULL, 0);
    int semid = semget(SEM_KEY, 1, 0666);
    
    // 初始化消息队列
    int msgid = msgget(MSG_KEY, IPC_CREAT | 0666);

    // 多进程并发处理
    for (int i = 0; i < WORKER_COUNT; i++) {
        pid_t pid = fork();
        if (pid == 0) {
            worker_process(i + 1, shm, semid, msgid);
            exit(0);
        }
    }

    // 父进程等待子进程 (监听 SIGCHLD 也可)
    while(wait(NULL) > 0);

    shmdt(shm);
    exit(0); // 替换原来的 return 0;
}

#ifndef LOGHAWK_H
#define LOGHAWK_H

#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/sem.h>
#include <sys/msg.h>

/* =========================================
 * 1. IPC 键值定义 (约定好的通信凭证)
 * ========================================= */
#define SHM_KEY 0x1234  // 共享内存 Key
#define SEM_KEY 0x5678  // 信号量 Key
#define MSG_KEY 0x9ABC  // 消息队列 Key

/* =========================================
 * 2. 系统核心配置常量
 * ========================================= */
#define BUFFER_SIZE 1024 // 日志单次读取/传输的最大缓冲区大小
#define WORKER_COUNT 3   // Processor 中的并发工作子进程数量
#define MAX_TOKENS 10    // Outputer 中令牌桶的最大令牌数 (流量控制上限)

/* =========================================
 * 3. 文件及路径定义
 * ========================================= */
#define LOG_FILE    "/var/log/app/source.log"        // 采集目标源文件
#define OUTPUT_FILE "/var/log/app/destination.log"   // 最终输出目标文件
#define LOCK_FILE   "/var/run/loghawk_outputer.pid"  // Outputer 单实例锁文件

/* =========================================
 * 4. 跨进程通信数据结构
 * ========================================= */
// 共享内存数据结构 (Collector -> Processor)
struct shared_data {
    char text[BUFFER_SIZE];
};

// 消息队列数据结构 (Processor -> Outputer)
struct msg_buffer {
    long msg_type;
    // 预留额外的 128 字节用于 Processor 追加时间戳、Worker ID 等元数据
    char msg_text[BUFFER_SIZE + 128]; 
};

/* =========================================
 * 5. 公共内联函数 (IPC 操作辅助)
 * ========================================= */
// 封装信号量的 P 操作 (申请资源/加锁)
static inline void sem_p(int semid) {
    struct sembuf sb = {0, -1, 0};
    semop(semid, &sb, 1);
}

// 封装信号量的 V 操作 (释放资源/解锁)
static inline void sem_v(int semid) {
    struct sembuf sb = {0, 1, 0};
    semop(semid, &sb, 1);
}


/* =========================================
 * 6. 模块入口函数声明
 * ========================================= */
void collector_main();
void processor_main();
void outputer_main();

#endif // LOGHAWK_H
